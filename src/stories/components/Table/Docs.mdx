## Features

<table>
  <thead>
    <tr>
      <th>Feature</th>
      <th>Description</th>
      <th>Enabled by</th>
    </tr>
  </thead>
  <tbody>
    <tr>
      <td>Sorting</td>

      <td>
        <code>sortable</code> on column
      </td>
    </tr>
    <tr>
      <td>Column Search</td>
      <td>Per-column search inputs</td>
      <td>
        <code>onColumnSearch</code> callback
      </td>
    </tr>
    <tr>
      <td>Column Filter</td>
      <td>Dropdown with checkboxes (Clear/Apply)</td>
      <td>
        <code>onColumnFilter</code> callback
      </td>
    </tr>
    <tr>
      <td>Global Search</td>
      <td>Search bar in toolbar</td>
      <td>
        <code>onTableSearch</code> callback
      </td>
    </tr>
    <tr>
      <td>Selection</td>
      <td>Checkbox column with select-all</td>
      <td>
        <code>selectable</code> attribute
      </td>
    </tr>
    <tr>
      <td>Row Click</td>
      <td>Pointer cursor + highlight on click</td>
      <td>
        <code>onRowClick</code> callback
      </td>
    </tr>
    <tr>
      <td>Disabled Rows</td>
      <td>Grey out rows, prevent interaction</td>
      <td>
        <code>_disabled</code> on row
      </td>
    </tr>
    <tr>
      <td>Frozen Columns</td>
      <td>Pin columns during horizontal scroll</td>
      <td>
        <code>frozen</code> on column
      </td>
    </tr>
    <tr>
      <td>Column Resizing</td>
      <td>Drag to resize, double-click to auto-fit</td>
      <td>
        <code>resizable</code> on column
      </td>
    </tr>
    <tr>
      <td>Column Configure</td>
      <td>Show/hide and freeze/unfreeze columns panel</td>
      <td>
        <code>column-configure</code> attribute
      </td>
    </tr>
    <tr>
      <td>Pagination</td>
      <td>Numbered pages or infinite scroll</td>
      <td>
        <code>pagination-type</code> attribute
      </td>
    </tr>
    <tr>
      <td>Nested Rows</td>
      <td>Expandable child rows</td>
      <td>
        <code>expandable</code> + <code>_children</code>

    </tr>
    <tr>
      <td>Export</td>

        <code>exportable</code> attribute

## Row Actions Example

Row actions add a kebab menu (three-dot icon) as the last column. Each row defines its own actions via `_actions`, or you can set a fallback via the `rowActions` property.

```typescript
interface ZetaTableAction {
  key: string; // Unique key passed to onRowAction callback
  label: string; // Display label
  icon?: string; // Optional icon
  disabled?: boolean; // Disable for a specific row
}
```

```javascript
table.data = [
  {
    id: 1,
    name: "Alice",
    role: "Admin",
    _actions: [
      { key: "edit", label: "Edit" },
      { key: "delete", label: "Delete" }
    ]
  },
  {
    id: 2,
    name: "Bob",
    role: "Viewer",
    _actions: [
      { key: "edit", label: "Edit" },
      { key: "delete", label: "Delete", disabled: true }
    ]
  }
];

table.onRowAction = (actionKey, row, rowIndex) => {
  console.log("Action:", actionKey, "on", row.name);
};
```

## How to use in ReactJs

> React &lt; 19 does not natively set JS object properties on custom elements — use a `ref`.

### Plain text cells

```tsx
import { useRef, useEffect } from "react";
import "@zebra-fed/zeta-web/dist/components/table/table.js";

const columns = [
  { field: "name", title: "Name", width: 180, sortable: true },
  { field: "email", title: "Email", width: 220 },
  { field: "role", title: "Role", width: 120 }
];

const users = [
  { id: 1, name: "Alice", email: "alice@acme.com", role: "Admin" },
  { id: 2, name: "Bob", email: "bob@acme.com", role: "Editor" },
  { id: 3, name: "Carol", email: "carol@acme.com", role: "Viewer" }
];

export function UsersTable() {
  const ref = useRef(null);

  useEffect(() => {
    const el = ref.current;
    if (!el) return;
    el.columns = columns;
    el.data = users;
  }, []);

  return (
    <zeta-table
      ref={ref}
      selectable
      show-data-count
      table-title="Users"
      pagination-type="numbered"
      page-size="5"
      style={{ "--table-row-height": "36px" } as React.CSSProperties}
    />
  );
}
```

#### Custom elements in cells

Use `ZetaTableReact` — a React wrapper that accepts JSX directly as cell values. It automatically renders React elements via portals, keeping them in the React tree (so context providers, state, event handlers all work).

> `ZetaTableReact` handles all the `ref`, `useLayoutEffect`, and portal plumbing internally.
> You just pass React components as cell values — no `document.createElement` needed.

```tsx
import { ZetaTableReact } from "@zebra-fed/zeta-web/react";

function StatusBadge({ status }: { status: string }) {
  return (
    <span
      style={{
        background: status === "active" ? "#d1fae5" : "#fee2e2",
        color: status === "active" ? "#065f46" : "#991b1b",
        padding: "2px 8px",
        borderRadius: "10px",
        fontSize: "12px"
      }}>
      {status}
    </span>
  );
}

const columns = [
  { field: "name", title: "Name", width: 180, sortable: true },
  { field: "email", title: "Email", width: 220 },
  { field: "status", title: "Status", width: 120 }
];

export function UsersTable() {
  const data = [
    { id: 1, name: "Alice", email: "alice@acme.com", status: <StatusBadge status="active" /> },
    { id: 2, name: "Bob", email: "bob@acme.com", status: <StatusBadge status="inactive" /> },
    { id: 3, name: "Carol", email: "carol@acme.com", status: <StatusBadge status="active" /> }
  ];

  return (
    <ZetaTableReact
      columns={columns}
      data={data}
      selectable
      showDataCount
      tableTitle="Users"
      paginationType="numbered"
      pageSize={5}
      onRowAction={(key, row) => console.log("Action:", key, row)}
      style={{ "--table-row-height": "36px" } as React.CSSProperties}
    />
  );
}
```

**Using `document.createElement` (without `ZetaTableReact`):**

If you prefer using the raw `<zeta-table>` element with a `ref`, you can pass DOM nodes:

```tsx
const tableData = users.map(user => {
  const badge = document.createElement("span");
  badge.textContent = user.status;
  badge.className = `badge badge-${user.status}`;

  return {
    id: user.id,
    name: user.name,
    status: badge // renders as an HTML element, not text
  };
});
```

**Using Angular components (via `ViewContainerRef`):**

```html
<!-- Add a hidden host for dynamic components -->
<ng-container #cellHost></ng-container>
```

```typescript
import { ViewChild, ViewContainerRef, ComponentRef, Type } from "@angular/core";

@ViewChild("cellHost", { read: ViewContainerRef }) cellHost!: ViewContainerRef;

private cellRefs: ComponentRef<any>[] = [];

private mapData(users: any[]): any[] {
  this.cellRefs.forEach((ref) => ref.destroy());
  this.cellRefs = [];

  return users.map((user) => ({
    id: user.id,
    name: user.name,
    status: this.renderCell(StatusBadgeComponent, { status: user.status }),
  }));
}

private renderCell(component: Type<any>, data: any): HTMLElement {
  const ref = this.cellHost.createComponent(component);
  Object.assign(ref.instance, data);
  ref.changeDetectorRef.detectChanges();
  this.cellRefs.push(ref);
  const el = ref.location.nativeElement as HTMLElement;
  el.style.display = "contents";
  return el;  // Angular component rendered inside the cell
}
```

**Using Vue components (via `render` + `h`):**

```vue
<template>
  <zeta-table
    :columns.prop="columns"
    :data.prop="tableData"
    selectable
    show-data-count
    table-title="Users"
    :style="{ '--table-row-height': '36px' }"
  />

<script setup>
import { computed, h, render, onUnmounted } from "vue";
import StatusBadge from "./StatusBadge.vue";

const columns = [
  { field: "name", title: "Name", width: 180, sortable: true },
  { field: "email", title: "Email", width: 220 },
  { field: "status", title: "Status", width: 120 },
];

const users = [
  { id: 1, name: "Alice", email: "alice@acme.com", status: "active" },
  { id: 2, name: "Bob",   email: "bob@acme.com",   status: "inactive" },
  { id: 3, name: "Carol", email: "carol@acme.com",  status: "active" },
];

const containers: HTMLElement[] = [];

function renderVueComponent(component, props) {
  const container = document.createElement("div");
  container.style.display = "contents";
  render(h(component, props), container);
  containers.push(container);
  return container;
}

const tableData = computed(() => {
  containers.forEach((el) => render(null, el));
  containers.length = 0;

  return users.map((user) => ({
    id: user.id,
    name: user.name,
    email: user.email,
    status: renderVueComponent(StatusBadge, { status: user.status }),
  }));
});

onUnmounted(() => {
  containers.forEach((el) => render(null, el));
});
</script>
```
